# 使用

你可以通过设置v-model来达到独立控制开关，或者控制开关的状态，activeColor inactiveColor分别控制开启和关闭的颜色

<template>
  <div id="app">
    <emui-switch name="12" v-model="active1"></emui-switch>

    <emui-switch
      name="11"
      v-model="active2"
      activeColor="red"
      inactiveColor="green"
    ></emui-switch>
  </div>
</template>

<script>
  export default {
    name: "App",
    data() {
      return {
        active1: false,
        active2: false,
      };
    },
  };
</script>
